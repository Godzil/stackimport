Sub ReadItemData( s as string )
  dim i as integer  dim d as string  Top=wasc(midb(s,&H2+1,2))  Left=wasc(midb(s,&H4+1,2))  Bottom=wasc(midb(s,&H6+1,2))  Right=wasc(midb(s,&H8+1,2))  Columns=wasc(midb(s,&HA+1,2))  i=wasc(midb(s,&HC+1,2))  ChangeTextHeight=(((i\8192) mod 2) = 1)  ChangeTextStyle=(((i\4096) mod 2) = 1)  ChangeTextSize=(((i\2048) mod 2) = 1)  ChangeTextFont=(((i\1024) mod 2) = 1)  Invert=(((i\16) mod 2) = 1)  RightFrame=(((i\8) mod 2) = 1)  BottomFrame=(((i\4) mod 2) = 1)  LeftFrame=(((i\2) mod 2) = 1)  TopFrame=(((i\1) mod 2) = 1)  TextSize=wasc(midb(s,&HE+1,2))  TextHeight=wasc(midb(s,&H10+1,2))  i=wasc(midb(s,&H12+1,2))  TextStyleGroup=(((i\32768) mod 2) = 1)  TextStyleExtend=(((i\16384) mod 2) = 1)  TextStyleCondense=(((i\8192) mod 2) = 1)  TextStyleShadow=(((i\4096) mod 2) = 1)  TextStyleOutline=(((i\2048) mod 2) = 1)  TextStyleUnderline=(((i\1024) mod 2) = 1)  TextStyleItalic=(((i\512) mod 2) = 1)  TextStyleBold=(((i\256) mod 2) = 1)  TextAlign=wasc(midb(s,&H14+1,2))  d=midb(s,&H16+1)  i=instrb(d,chrb(0))  Contents=leftb(d,i-1)  TextFont=midb(d,i+1)  if instrb(TextFont,chrb(0))>0 then    i=instrb(TextFont,chrb(0))    TextFont=leftb(TextFont,i-1)  end if
