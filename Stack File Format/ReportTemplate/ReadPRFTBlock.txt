Sub ReadPRFTBlock( s as string )
  dim d as string  dim i as integer  dim numItems as integer  dim itemLength as integer  dim itemData as string  dim ri as ReportItem  ID=wasc(midb(d,&H2+1,2))  Units=wasc(midb(d,&H8+1,2))\256  TopMargin=wasc(midb(d,&HA+1,2))  LeftMargin=wasc(midb(d,&HC+1,2))  BottomMargin=wasc(midb(d,&HE+1,2))  RightMargin=wasc(midb(d,&H10+1,2))  SpacingHeight=wasc(midb(d,&H12+1,2))  SpacingWidth=wasc(midb(d,&H14+1,2))  CellHeight=wasc(midb(d,&H16+1,2))  CellWidth=wasc(midb(d,&H18+1,2))  i=wasc(midb(d,&H1A+1,2))  LeftToRight=(((i\256) mod 2)=1)  DynamicHeight=(((i\256) mod 2)=1)  i=ascb(midb(d,&H1C+1,1))  Header=midb(d,&H1C+2,i)  d=midb(s,&H11C+1)  numItems=wasc(leftb(d,2))  d=midb(d,3)  if numItems>0 then    for i=1 to numItems      itemLength=wasc(leftb(d,2))      itemData=midb(d,3,itemLength)      ri = new ReportItem      ri.ReadItemData itemData      Itm.append ri      if (itemLength mod 2)=1 then        itemLength=itemLength+1      end if      d=midb(d,itemLength+1)    next  end if
