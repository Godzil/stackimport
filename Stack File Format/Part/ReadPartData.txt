Sub ReadPartData( s as string )
  dim i as integer  dim d as string  ID=wasc(midb(s,&H2+1,2))  i=wasc(midb(s,&H4+1,2))  PartType=(i\256)  Visible=(((i\128) mod 2) = 0)  DontWrap=(((i\32) mod 2) = 1)  DontSearch=(((i\16) mod 2) = 1)  SharedText=(((i\8) mod 2) = 1)  FixedLineHeight=(((i\4) mod 2) = 0)  AutoTab=(((i\2) mod 2) = 1)  Enabled=(((i\1) mod 2) = 0)  LockText=(((i\1) mod 2) = 1)  Top=wasc(midb(s,&H6+1,2))  Left=wasc(midb(s,&H8+1,2))  Bottom=wasc(midb(s,&HA+1,2))  Right=wasc(midb(s,&HC+1,2))  i=wasc(midb(s,&HE+1,2))  Style=(i mod 16)  i=i\256  ShowName=(((i\128) mod 2) = 1)  AutoSelect=(((i\128) mod 2) = 1)  Hilite=(((i\64) mod 2) = 1)  ShowLines=(((i\64) mod 2) = 1)  AutoHilite=(((i\32) mod 2) = 1)  WideMargins=(((i\32) mod 2) = 1)  SharedHilite=(((i\16) mod 2) = 0)  MultipleLines=(((i\16) mod 2) = 1)  Family=(i mod 16)  TitleWidth=wasc(midb(s,&H10+1,2))  LastSelectedLine=wasc(midb(s,&H10+1,2))  Icon=wasc(midb(s,&H12+1,2))  SelectedLine=wasc(midb(s,&H12+1,2))  TextAlign=wasc(midb(s,&H14+1,2))  TextFontID=wasc(midb(s,&H16+1,2))  TextSize=wasc(midb(s,&H18+1,2))  i=wasc(midb(s,&H1A+1,2))  TextStyleGroup=(((i\32768) mod 2) = 1)  TextStyleExtend=(((i\16384) mod 2) = 1)  TextStyleCondense=(((i\8192) mod 2) = 1)  TextStyleShadow=(((i\4096) mod 2) = 1)  TextStyleOutline=(((i\2048) mod 2) = 1)  TextStyleUnderline=(((i\1024) mod 2) = 1)  TextStyleItalic=(((i\512) mod 2) = 1)  TextStyleBold=(((i\256) mod 2) = 1)  TextHeight=wasc(midb(s,&H1C+1,2))  d=midb(s,&H1E+1)  i=instrb(d,chrb(0))  Name=leftb(d,i-1)  Script=midb(d,i+1)  if rightb(Script,1)=chrb(0) and leftb(Script,1)<>chrb(0) then    Script=leftb(Script,lenb(Script)-1)  end if
